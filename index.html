Question 1
Separate odd and even numbers in a list
Given a list of integers, write a Java 8 program to separate the odd and even numbers into two separate lists.
Java 8 Solution
import java.util.*;
import java.util.stream.*;

public class OddEven {
    public static void main(String[] args) {
        List<Integer> numbers = Arrays.asList(10, 15, 20, 25, 30);

        List<Integer> even = numbers.stream()
                .filter(n -> n % 2 == 0)
                .collect(Collectors.toList());

        List<Integer> odd = numbers.stream()
                .filter(n -> n % 2 != 0)
                .collect(Collectors.toList());

        System.out.println("Even: " + even);
        System.out.println("Odd: " + odd);
    }
}

Question 22
Find the most repeated element in an array
Write a Java 8 program to find the most repeated element in an array.
Java 8 Solution
import java.util.*;
import java.util.function.Function;
import java.util.stream.Collectors;

public class MostRepeated {
    public static void main(String[] args) {
        Integer[] arr = {1, 2, 3, 2, 4, 2, 5, 3, 3};

        Map<Integer, Long> freq = Arrays.stream(arr)
                .collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));

        int mostRepeated = freq.entrySet().stream()
                .max(Map.Entry.comparingByValue())
                .get()
                .getKey();

        System.out.println("Most Repeated: " + mostRepeated);
    }
}

Question 32
Calculate age in years
Write a Java 8 program to calculate the age of a person in years using their date of birth.
Java 8 Solution
import java.time.LocalDate;
import java.time.Period;

public class CalculateAge {
    public static void main(String[] args) {
        LocalDate dob = LocalDate.of(1999, 5, 20);
        LocalDate currentDate = LocalDate.now();

        int age = Period.between(dob, currentDate).getYears();

        System.out.println("Age: " + age);
    }
}
